自定义地图

var map = new BMap.Map("container");
var point = new BMap.Point(119.877986, 40.026843);
map.centerAndZoom(point, 19);
map.setMapStyleV2({
    // 自定义地图id
    styleId: '86fa69b2662e96d275711bc5c591283b'
});
// 开启鼠标滚轮
map.enableScrollWheelZoom(true);

// 创建点覆盖物
var myIcon = new BMap.Icon("static/images/zuobiao.png", new BMap.Size(87, 93));
var marker1 = new BMap.Marker(new BMap.Point(119.878986, 40.026843));
marker1.setIcon(myIcon);
marker1.setAnimation(BMAP_ANIMATION_BOUNCE);
map.addOverlay(marker1);

// 自定义文本标注
var opts = {
    // 指定文本标注所在的地理位置
    position: point,
    // 设置文本偏移量
    offset: new BMap.Size(30, -30)
}
var label = new BMap.Label(
        "<div class=\"text\">" +
            "<p>我是自定义字段</p>" +
        "</div>"
        // 或者
        $('template').html()
    , opts);
// 创建文本标注对象
map.addOverlay(label);

// 坐标变换
var this_point = $(this).data('point').split(','); // 获取当前点坐标
var point = new BMap.Point(parseFloat(this_point[0]),parseFloat(this_point[1]));
map.setCenter(point);

var coordinate=new google.maps.LatLng(34.756307129471125, 135.50143301160293);
function initialize() {
    var mapProp = {
        center:coordinate,
        zoom:7,
        mapTypeId:google.maps.MapTypeId.ROADMAP
    };
    var map=new google.maps.Map(document.getElementById("googleMap"),mapProp);
    var marker=new google.maps.Marker({
        position:coordinate,
    });
    marker.setMap(map);
    var infowindow = new google.maps.InfoWindow({
        content: "Hello"
    });
    google.maps.event.addListener(marker, 'click', function() {
        infowindow.open(map,marker);
    });
}
google.maps.event.addDomListener(window, 'load', initialize);
