JS 代码:
const scroll = new LocomotiveScroll({
    el: document.querySelector('.js_scroll'),
    smooth: true,
    direction: 'horizontal', // 水平滚动
});
// 监听LocomotiveScroll的scroll事件
scroll.on('scroll', function (scroll) {
    var scrollTop = scroll.scroll.y
    console.log(scrollTop);
});
