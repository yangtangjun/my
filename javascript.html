预加载img
JS 代码:

加载页面所有的img标签的图片

setTimeout(()=>{
    var load=0
    var arr=new Array();
    $('img').each(function (index,ele){
        if($(this).attr('src')!==''){
            arr.push($(this).attr('src'))
        }
    })
    var len=arr.length
    for(var i in arr){
        var bimg=new Image();
        bimg.src = arr[i];
        bimg.onload = function () {
            load+=1;
            $('div').css({
                width:load/len*100+'%',
            })
            if(len===load){
                console.log(111)
            }
        }
    }
},10)

或者

setTimeout(() => {
    let imgList = document.querySelectorAll('img');
    let imgSrcList = [];

    for (let i = 0; i < imgList.length; i++) {
        var imgSrc = imgList[i].getAttribute('src');
        if (imgSrc) {
            imgSrcList.push(imgSrc);
        }
    }

    let len = imgSrcList.length;
    let loadedCount = 0;
    let progress = 0;

    function updateProgress (){
        document.querySelector('.load .num').innerHTML = `${progress}%`;
    }

    function incrementProgress ()  {
        if (progress < loadedCount / len * 100) {
            progress++;
            updateProgress();
        }
    }

    for (let i = 0; i < len; i++) {
        const bimg = new Image();
        bimg.src = imgSrcList[i];
        bimg.onload = function() {
            loadedCount++;
            if (loadedCount === len) {
                console.log('图片加载完毕')
            }
        };
    }

    setInterval(incrementProgress, 10);
}, 10);

监听鼠标滚轮事件

if (window.addEventListener)
    window.addEventListener('DOMMouseScroll', wheel, false);
window.onmousewheel = document.onmousewheel = wheel;
function wheel(event){
    var delta = 0;
    if (!event) event = window.event;
    if (event.wheelDelta) {
        delta = event.wheelDelta/120;
        if (window.opera) delta = -delta;
    } else if (event.detail) {
        delta = -event.detail/3;
    }
    if (delta)
        handle(delta);
}
function handle(delta) {
if (delta <0){
        //向下滚动
        console.log('下！')
    }else{
        //向上滚动
        console.log('上！')
    }
}
